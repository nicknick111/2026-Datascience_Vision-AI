# MySQL 데이터베이스 접속 정보 (내 PC에 설치된 DB 기준)
DB_HOST = "127.0.0.1"  # 127.0.0.1은 '내 컴퓨터(localhost)'를 의미합니다.
DB_USER = "root"       # MySQL 관리자 계정 아이디
DB_PASS = "1234"       # MySQL 관리자 계정 비밀번호
DB_NAME = "dronedb"    # 우리가 생성하고 사용할 데이터베이스의 이름

# 추출된 이미지 정보를 실시간으로 보낼 외부 API 서버의 주소
API_ENDPOINT = "http://127.0.0.1:8080/api/training-data"

# 무인기 부품 식별을 위한 17개의 '사용자 정의 클래스(분야)' 목록
# YOLO 모델은 문자를 모르기 때문에 0번(UAV_Wheel)부터 16번(UAV_Fuselage_1)까지 숫자로 매핑하여 학습합니다.
CLASS_NAMES = [
    "UAV_Wheel", "Left_wing", "Right_wing", "Fuselage_Part_1", "Fuselage_Part_2_1", 
    "Fuselage_Part_2_2", "Fuselage_Part_3_1", "Fuselage_Part_3_2", "Fuselage_Part_4_1", 
    "ISR_System_1", "ISR_System_2", "Jet_Engine_1", "Jet_Engine_2", 
    "Missile_Launcher", "Missile", "Pylon", "UAV_Fuselage_1"
]

# [속도 최적화] API 전송 전담 일꾼 10명(스레드 풀)을 미리 대기시킵니다.
# 메인 프로그램이 영상을 자르느라 바쁠 때, 이 일꾼들이 뒤에서 조용히 API 서버로 데이터를 보냅니다.
api_executor = concurrent.futures.ThreadPoolExecutor(max_workers=10)


def setup_local_database():
    """
    [기능] 내 PC에 MySQL 데이터베이스(dronedb)와 데이터를 담을 테이블을 자동으로 생성합니다.
    [용도] 사용자가 터미널에서 복잡한 SQL 명령어를 칠 필요 없이 파이썬 실행만으로 DB 환경을 세팅하기 위함.
    """
    print("\n" + "=" * 60)
    print("🚀 [목표 1] 내 PC 데이터베이스 및 테이블 자동 생성 (최적화 버전)")
    print("=" * 60)
    
    # 1. 데이터베이스(dronedb) 생성 파트
    try:
        # 데이터베이스 지정 없이 MySQL 서버 자체에만 먼저 접속합니다.
        conn = pymysql.connect(host=DB_HOST, user=DB_USER, password=DB_PASS, port=3306)
        cur = conn.cursor()
        # 만약 'dronedb'가 없다면 만들어라! 라는 SQL 명령입니다.
        cur.execute(f"CREATE DATABASE IF NOT EXISTS {DB_NAME}")
        conn.commit()
        conn.close()
        print(f"  🟢 [DB 생성] '{DB_NAME}' 데이터베이스 확보 완료")
    except Exception as e:
        print(f"  🔴 [DB 에러] MySQL 서버 접속 실패: {e}")
        return False # 접속 실패 시 프로그램 중단을 위해 False 반환

    # 2. 테이블(training_dataset) 생성 파트
    try:
        # 이번엔 생성된 'dronedb' 데이터베이스 안으로 직접 접속합니다.
        conn = pymysql.connect(host=DB_HOST, user=DB_USER, password=DB_PASS, database=DB_NAME, port=3306)
        cur = conn.cursor()
        # 학습 데이터의 이미지 경로, 클래스 이름, 크기, 저장 시간을 기록할 표(Table)를 만듭니다.
        cur.execute("""
            CREATE TABLE IF NOT EXISTS training_dataset (
                id INT AUTO_INCREMENT PRIMARY KEY,          -- 데이터 고유 번호 (자동 증가)
                image_path TEXT NOT NULL,                   -- PC에 저장된 이미지의 물리적 절대 경로
                class_id INT,                               -- 0~16의 정수 클래스 번호
                class_name VARCHAR(50),                     -- "UAV_Wheel" 같은 문자열 클래스 이름
                width INT,                                  -- 이미지 가로 길이 (기본 640)
                height INT,                                 -- 이미지 세로 길이 (기본 640)
                split VARCHAR(10) DEFAULT 'train',          -- 용도 (train: 학습용)
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP -- DB에 데이터가 저장된 현재 시간
            );
        """)
        conn.commit() # 변경사항 확정 저장
        conn.close()
        print("  🟢 [테이블 생성] 'training_dataset' 테이블 준비 완료")
        return True
    except Exception as e:
        print(f"  🔴 [DB 에러] 테이블 생성 실패: {e}")
        return False


def check_api_server():
    """
    [기능] 지정된 API 서버(8080 포트)가 살아서 응답하는지 똑똑 노크(GET 요청)해봅니다.
    [용도] 본격적인 작업 전, API 서버가 꺼져있는지 미리 파악하여 사용자에게 알려주기 위함.
    """
    print("\n🚀 [목표 2] API 서버 연결 점검")
    try:
        # 2초 동안만 대답을 기다려봅니다. (timeout=2)
        res = requests.get(API_ENDPOINT, timeout=2)
        print(f"  🟢 [API 정상] 서버 통신 가능 (응답: {res.status_code})")
    except Exception as e:
        # API 서버가 꺼져있어도 프로그램이 다운되지 않고 단순히 경고만 띄우고 넘어갑니다. (비동기 처리 덕분)
        print(f"  🔴 [API 경고] API 서버에 접속할 수 없습니다 (비동기 처리로 무시하고 진행됨).")


def send_api_async(payload):
    """
    [기능] 추출된 이미지 정보를 API 서버로 POST 전송합니다.
    [용도] ThreadPoolExecutor(스레드 풀)에 의해 백그라운드에서 실행되어, 영상 추출의 메인 속도를 저하시키지 않습니다.
    """
    try:
        # 1초 이상 응답이 없으면 그냥 통신을 포기하고 다음 작업으로 넘어갑니다. (속도 방어 목적)
        requests.post(API_ENDPOINT, json=payload, timeout=1)
    except:
        pass # 통신에 실패해도 콘솔에 에러를 띄우지 않고 조용히 무시합니다.

# [목표 3 & 4] 영상 추출, 전처리, DB/API 연동 (초고속 최적화)
# ==========================================
def extract_and_preprocess(video_path, class_id, class_name, output_dir="datasets/images/train", step=5):
    """
    [기능] mp4 영상을 불러와 step(간격)마다 이미지를 1장씩 자르고(추출), 
           YOLO가 좋아하는 640x640 크기로 리사이징(전처리)한 후, 데이터를 모아 DB에 한 번에 넣습니다.
    [용도] 날것의 영상 데이터를 AI가 씹어 먹을 수 있는 최적의 학습 재료(이미지)로 변환하는 핵심 공정.
    """
    # 물리적으로 해당 mp4 파일이 존재하는지 검사
    if not os.path.exists(video_path):
        print(f"  [-] 누락: '{video_path}' 파일을 찾을 수 없습니다.")
        return

    # 이미지를 저장할 폴더(datasets/images/train)가 없으면 생성
    os.makedirs(output_dir, exist_ok=True)
    
    # OpenCV를 이용해 영상 파일을 엽니다.
    cap = cv2.VideoCapture(video_path)
    saved = 0 # 현재까지 저장된 이미지 장수 카운트
    
    # [DB 속도 최적화] 이미지가 잘릴 때마다 매번 DB에 넣으면 속도가 너무 느려집니다.
    # 따라서 db_records 라는 빈 바구니(리스트)를 만들어서 데이터를 모아만 둡니다.
    db_records = []

    print(f"\n▶ [{class_name}] 초고속 영상 추출 및 전처리 시작...")
    
    while True:
        # 1. 영상에서 1프레임(사진 1장)을 읽어와서 디코딩(해석)합니다.
        ret, frame = cap.read()
        if not ret: break # 더 이상 읽을 영상 프레임이 없으면 반복문 종료

        # 2. [전처리] YOLOv8 모델이 가장 빠르고 정확하게 학습하는 기본 해상도인 640x640으로 크기를 강제 조정합니다.
        frame_resized = cv2.resize(frame, (640, 640))
        
        # 3. 고유한 파일명 생성: '클래스명_현재시간_0001.jpg' 형식으로 이름이 겹치지 않게 만듭니다.
        filename = os.path.join(output_dir, f"{class_name}_{datetime.now().strftime('%Y%m%d%H%M%S')}_{saved:04d}.jpg")
        cv2.imwrite(filename, frame_resized) # 완성된 이미지를 하드디스크에 실제로 저장
        abs_path = os.path.abspath(filename) # DB에 기록하기 위해 방금 저장한 이미지의 전체 절대 경로를 가져옴
        
        # 4. 방금 저장한 이미지의 각종 정보를 db_records 바구니에 담습니다. (나중에 한 번에 쏟아부을 예정)
        db_records.append((abs_path, class_id, class_name, 640, 640))
        
        # 5. [API 속도 최적화] API 서버로 보낼 데이터를 조립한 뒤, 
        # 메인 프로그램이 직접 보내지 않고, 아까 고용한 'api_executor' 일꾼에게 던져버리고 바로 넘어갑니다.
        payload = {"image_path": abs_path, "class_id": class_id, "class_name": class_name, "resolution": "640x640", "timestamp": datetime.now().isoformat()}
        api_executor.submit(send_api_async, payload)
        
        saved += 1
        # 터미널 창이 지저분해지지 않게 현재 저장 진행 상황을 한 줄에서만 깜빡이며 업데이트(\r) 합니다.
        print(f"  - {saved:04d}장 추출 완료 (데이터 적재 중...)", end="\r")

        # 6. [연산 최적화] 우리가 step=5로 설정했다면, 1장 저장 후 4장은 그냥 버려야 합니다.
        # cap.read()로 읽으면 버릴 이미지까지 화질을 분석하느라 CPU를 낭비합니다.
        # cap.grab()을 사용하면 화질 분석(디코딩) 없이 비디오 테이프를 빨리 감기하듯 휙휙 건너뛰어 엄청나게 빨라집니다.
        for _ in range(step - 1):
            if not cap.grab():
                break

    # 메모리 누수 방지를 위해 열었던 영상 파일을 닫아줍니다.
    cap.release()
    print(f"\n  ✅ [{class_name}] 총 {saved}장 이미지 추출 완료.")
    
    # 7. [DB 속도 최적화] 영상 1개가 모두 끝나고 닫히면, 
    # db_records 바구니에 모아뒀던 수백~수천 개의 데이터를 DB에 단 1번의 통신으로 들이붓습니다(Bulk Insert).
    if db_records:
        try:
            print("  ⏳ [DB 일괄 저장] 추출된 데이터를 데이터베이스에 기록 중입니다...")
            conn = pymysql.connect(host=DB_HOST, user=DB_USER, password=DB_PASS, database=DB_NAME, port=3306)
            cur = conn.cursor()
            # executemany 함수가 핵심입니다. SQL 쿼리 하나에 리스트 전체를 매핑하여 한 번에 삽입합니다.
            cur.executemany("""
                INSERT INTO training_dataset (image_path, class_id, class_name, width, height) 
                VALUES (%s, %s, %s, %s, %s)
            """, db_records)
            conn.commit()
            conn.close()
            print("  🟢 [DB 저장 완료] 벌크 인서트(Bulk Insert) 성공!")
        except Exception as e:
            print(f"  🔴 [DB 에러] 일괄 저장 실패: {e}")

# [목표 5] YOLOv8n AI Model 학습 (Train)

def train_yolo_model():
    """
    [기능] 추출 및 라벨링이 완료된 데이터셋을 불러와 YOLOv8n 모델에게 학습을 시킵니다.
    [용도] 단순한 이미지 모음을 넘어, 컴퓨터가 무인기 부품을 식별할 수 있는 '지능(Weight)'을 만들어내는 최종 단계.
    """
    print("\n" + "=" * 60)
    print("🚀 [목표 5] 변환된 데이터로 YOLOv8n 커스텀 학습 시작")
    print("=" * 60)
    
    # 1. YOLOv8은 학습을 위해 데이터가 어디에 있고 클래스가 몇 개인지 적힌 YAML 환경 설정 파일을 반드시 요구합니다.
    yaml_path = os.path.abspath("datasets/drone_parts.yaml")
    data_config = {
        "path": os.path.abspath("datasets"),
        "train": "images/train",  # 모델이 공부할 이미지가 있는 폴더
        "val": "images/train",    # 모델이 스스로 채점(검증)해볼 이미지가 있는 폴더
        "nc": len(CLASS_NAMES),   # 총 클래스(부품) 개수 (현재 17개)
        "names": CLASS_NAMES      # 0~16번에 해당하는 부품 이름 배열
    }
    # 파이썬 딕셔너리를 yaml 파일로 물리적으로 변환하여 저장합니다.
    with open(yaml_path, "w", encoding="utf-8") as f:
        yaml.dump(data_config, f, allow_unicode=True, sort_keys=False)
    
    # 2. 아주 가볍고 빠른 최신 나노(nano) 모델인 'yolov8n.pt'의 기초 지식을 불러옵니다.
    model = YOLO("yolov8n.pt")
    
    print("\n[INFO] 다각도(회전/원근) 인식 보정 파라미터가 적용되어 학습됩니다.")
    
    # 3. 본격적인 학습을 명령하는 train() 함수입니다. 여기에 적용된 파라미터들이 모델 성능을 좌우합니다.
    model.train(
        data=yaml_path,
        epochs=100,        # [반복] 보유한 전체 데이터셋을 총 100번 반복해서 꼼꼼하게 복습시킵니다.
        imgsz=640,         # [해상도] 우리가 전처리한 크기인 640으로 맞춰 모델에 입력합니다.
        batch=16,          # [배치] 한 번에 그래픽카드(RAM)에 올려서 처리할 이미지 장수 (PC 성능에 따라 8, 16, 32 조정)
        patience=20,       # [조기 종료] 20번 넘게 반복해도 똑똑해지지 않으면(오차가 안 줄면) 시간낭비 방지를 위해 100번 전에 조기 종료시킵니다.
        
        # --- [데이터 증강 (Data Augmentation) 파라미터] ---
        # ※ 이전에 원기둥은 인식되나 육면체가 인식 안 되던 문제를 해결하는 핵심 마법의 코드입니다.
        degrees=45.0,      # 이미지를 임의로 최대 45도까지 회전시킵니다. (비스듬히 놓인 부품도 인식 가능하게 함)
        perspective=0.001, # 렌즈의 원근감을 임의로 왜곡시킵니다. (가까이서 찍거나 멀리서 찍은 듯한 3D 효과 부여)
        flipud=0.3,        # 30% 확률로 이미지를 상하로 뒤집습니다.
        fliplr=0.5,        # 50% 확률로 이미지를 좌우로 뒤집습니다.
        
        project="runs/train",        # 학습 결과물(로그, 그래프 등)이 저장될 최상위 폴더 이름
        name="drone_quality_model"   # 현재 진행하는 학습 프로젝트의 고유 이름 지정
    )
    print("\n🎉 [완료] 최적화된 모델 가중치가 'runs/train/drone_quality_model/weights/best.pt'에 저장되었습니다!")
