import os
# OpenCV가 내부적으로 출력하는 불필요한 경고 메시지나 로그를 숨겨서 콘솔창을 깨끗하게 유지합니다.
os.environ["OPENCV_LOG_LEVEL"] = "SILENT"
os.environ["OPENCV_VIDEOIO_DEBUG"] = "0"

import cv2          # 영상 처리의 핵심인 OpenCV 라이브러리입니다.
import time         # FPS(초당 프레임 수)를 계산하거나 시간을 측정할 때 사용합니다.
import datetime     # 현재 날짜와 시간을 생성하여 서버에 보낼 때 사용합니다.
import numpy as np  # 행렬 연산(이미지를 자르고 붙이고 숫자를 더하는 작업)을 아주 빠르게 해주는 라이브러리입니다.
import requests     # 인터넷(네트워크)을 통해 API 서버로 데이터를 보낼 때 사용하는 통신 라이브러리입니다.
import threading    # [최적화 핵심] 메인 화면이 멈추지 않도록, 백그라운드에서 심부름을 대신 해줄 '스레드(일꾼)' 라이브러리입니다.

# =========== 서버 연결 부분 ============
# 통신할 API 서버의 주소입니다. (현재 사내망 또는 로컬 네트워크 IP로 설정됨)
API_SERVER = "http://192.168.3.21:8000"

def send_inspection_result(result: str):
    """
    판정 결과를 API 서버로 전송하는 함수입니다.
    (주의: 화면이 끊기지 않도록 이제 이 함수는 메인 흐름이 아닌 '백그라운드 스레드'에서 몰래 실행됩니다.)
    """
    try:
        # 서버에서 요구하는 형식(JSON)에 맞추어 데이터를 포장합니다.
        data = {
            "product_id": f"DRONE-{datetime.datetime.now().strftime('%Y%m%d%H%M%S')}", # 예: DRONE-20260324103000
            "result": result,                                                          # 판정 결과 (예: "정상")
            "timestamp": datetime.datetime.now().isoformat(),                          # 현재 전송 시간
            "camera_id": "RPC-20F"                                                     # 작업 중인 카메라 장비명
        }
        # requests.post를 이용해 서버로 데이터를 쏩니다.
        response = requests.post(f"{API_SERVER}/inspection", json=data)
        print(f"[API] 전송 완료: {result} → 상태코드: {response.status_code}")
    except Exception as e:
        # 서버가 꺼져있거나 인터넷이 끊겼을 때 프로그램이 튕기지 않도록 에러를 잡아줍니다.
        print(f"[API ERROR] 전송 실패: {e}") 
# =======================================

# 프로그램 전체에서 공통으로 상태를 기억하기 위한 전역 변수들입니다.
is_paused = False        # 일시정지 상태인지 확인 (True면 화면 멈춤)
is_running = True        # 프로그램이 실행 중인지 확인 (False면 프로그램 종료)
use_glare_filter = True  # 화면 빛 반사/그림자 억제 필터(CLAHE)의 기본 켜짐/꺼짐 상태
cap = None               # 카메라 장치(객체)를 담아둘 빈 공간입니다.

def mouse_callback(event, x, y, flags, param):
    """
    마우스 클릭을 감지하는 함수입니다.
    사용자가 캔버스(도화지)의 특정 위치를 클릭하면, 그 좌표(x, y)를 확인해서 알맞은 버튼 기능을 실행합니다.
    """
    global is_paused, is_running
    # 마우스 왼쪽 버튼(LBUTTONDOWN)이 눌렸을 때만 작동합니다.
    if event == cv2.EVENT_LBUTTONDOWN:
        # 버튼들이 모여있는 Y축 높이 범위 (하단 UI 패널 영역)
        if 680 <= y <= 750:
            if 40 <= x <= 180:      
                is_paused = False # PLAY 영역 클릭 -> 일시정지 해제
            elif 220 <= x <= 360:   
                is_paused = True  # PAUSE 영역 클릭 -> 일시정지 작동
            elif 400 <= x <= 540:   
                is_running = False # EXIT 영역 클릭 -> 프로그램 종료 신호 보냄

def on_brightness_trackbar(val):
    """
    소프트웨어 밝기 조절 바(Trackbar)를 움직일 때마다 호출되는 함수입니다.
    OpenCV의 문법 규칙상 빈 함수라도 반드시 만들어 두어야 에러가 나지 않습니다.
    """
    pass

def on_exposure_trackbar(val):
    """
    카메라 렌즈의 하드웨어 노출값을 물리적으로 조절하는 핵심 함수입니다.
    창문이나 형광등의 강력한 눈부심(빛 날아감)을 근본적으로 억제합니다.
    """
    global cap
    if cap is not None and cap.isOpened():
        # 트랙바는 0~10의 숫자만 줍니다. 하지만 카메라는 음수(-10 ~ 0)를 원합니다.
        # 따라서 받아온 값에서 10을 빼서 진짜 노출값(-10 ~ 0)으로 둔갑시킵니다.
        exposure_val = val - 10 
        cap.set(cv2.CAP_PROP_EXPOSURE, exposure_val) # 카메라 렌즈에 진한 선글라스를 씌우거나 벗깁니다.

def apply_clahe(image):
    """
    CLAHE (국소적 대비 평활화) 필터입니다. 
    단순히 밝기만 올리는 것이 아니라, 화면을 잘게 쪼개어 빛이 강한 곳(창문)은 누르고, 
    어두운 곳(기계 그늘)은 밝게 끄집어내어 전체적인 디테일을 살려주는 고급 기술입니다.
    """
    # 빛과 색상을 분리하기 위해 BGR(일반 이미지)를 LAB 색상 공간으로 바꿉니다.
    lab = cv2.cvtColor(image, cv2.COLOR_BGR2LAB)
    l, a, b = cv2.split(lab) # l: 밝기(빛), a: 빨강-초록 색상, b: 노랑-파랑 색상
    
    # 밝기(l) 채널에만 필터를 씌워서 색상이 이상하게 변하는 것을 막습니다. (clipLimit 2.0이 자연스러움)
    clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8, 8))
    cl = clahe.apply(l)
    
    # 필터가 적용된 밝기 채널과 원래 색상 채널을 다시 합칩니다.
    limg = cv2.merge((cl, a, b))
    final = cv2.cvtColor(limg, cv2.COLOR_LAB2BGR) # 다시 우리가 보는 일반(BGR) 이미지로 돌려놓습니다.
    return final

def main():
    global is_paused, is_running, use_glare_filter, cap
    # 프로그램 실행 시 터미널(콘솔) 창을 깨끗하게 지워줍니다. (nt는 윈도우, 그 외는 리눅스/맥)
    os.system('cls' if os.name == 'nt' else 'clear')
    
    # 0번(기본) 카메라를 연결합니다. CAP_DSHOW는 윈도우 환경에서 카메라를 빠르게 켜기 위한 옵션입니다.
    cap = cv2.VideoCapture(0, cv2.CAP_DSHOW)
    if not cap.isOpened():
        print("[ERROR] Camera not found.")
        return

    # 카메라 하드웨어의 화질(해상도)을 HD급(1280x720)으로 설정합니다.
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 1280)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 720)
    # [매우 중요] 우리가 노출(밝기)을 수동으로 조절하기 위해 카메라가 스스로 밝기를 바꾸는 '자동 노출' 기능을 강제로 끕니다.
    cap.set(cv2.CAP_PROP_AUTO_EXPOSURE, 0.25) 

    # --- [최적화 3] 캘리브레이션 픽셀 지도(Map) 미리 만들어두기 ---
    has_calibration = False
    map1, map2 = None, None
    try:
        # 이전에 찍어둔 체스보드 왜곡 보정 데이터를 불러옵니다.
        calib_data = np.load('calibration.npz')
        mtx = calib_data['mtx']   # 카메라 고유의 렌즈 매트릭스
        dist = calib_data['dist'] # 얼마나 휘었는지 알려주는 왜곡 계수
        has_calibration = True
        
        # 현재 카메라가 찍고 있는 화면의 가로, 세로 크기를 가져옵니다.
        w = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
        h = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
        
        # [핵심 속도 향상] 매 프레임 수식을 계산해 픽셀을 펴는 건 너무 느립니다. (undistort의 단점)
        # 대신, '원래 (x, y)에 있던 픽셀이 보정 후에는 어디로 가야 하는지'를 미리 계산해 거대한 지도(map1, map2)로 만들어 둡니다.
        map1, map2 = cv2.initUndistortRectifyMap(mtx, dist, None, mtx, (w, h), cv2.CV_16SC2)
        print("[INFO] Calibration data loaded and map generated successfully.")
    except:
        # 데이터 파일이 없으면 보정 없이 원본 그대로 진행합니다.
        print("[WARN] No calibration.npz found. Skipping lens correction.")

    # 시작 안내 메시지 출력
    print("=" * 60)
    print("[SYSTEM] RPC-20F Vision Studio v2.9 (High FPS Optimized) is running.")
    print("=" * 60)

    # 모니터링할 창(Window)을 만들고, 앞서 만든 마우스 감지 함수를 이 창에 연결합니다.
    window_name = 'RPC-20F Vision Studio v2.9'
    cv2.namedWindow(window_name)
    cv2.setMouseCallback(window_name, mouse_callback)

    # 창의 맨 위에 트랙바(슬라이더)를 생성합니다.
    cv2.createTrackbar('S/W Brightness (-100~100)', window_name, 100, 200, on_brightness_trackbar)
    cv2.createTrackbar('Lens Exposure (-10~0)', window_name, 2, 10, on_exposure_trackbar)
    # 프로그램 시작 시 렌즈 노출이 너무 밝지 않게, 찾으신 최적값인 -8(트랙바 눈금 2)로 즉시 설정합니다.
    on_exposure_trackbar(2)

    # --- [최적화 2] 변하지 않는 UI 배경을 미리 한 번만 그려서 도장(Cache)으로 만들기 ---
    # 도화지 크기(798x1152)의 거대한 캔버스를 만듭니다. (3은 컬러 이미지를 의미)
    base_canvas = np.zeros((798, 1152, 3), dtype=np.uint8)
    # 캔버스 하단(y: 648~798)을 UI 전용 어두운 회색 배경으로 칠합니다.
    cv2.rectangle(base_canvas, (0, 648), (1152, 798), (35, 35, 35), -1)
    
    # PLAY 버튼 그리기
    cv2.rectangle(base_canvas, (40, 680), (180, 750), (0, 200, 0), -1)
    cv2.putText(base_canvas, "PLAY", (75, 720), cv2.FONT_HERSHEY_SIMPLEX, 0.8, (255, 255, 255), 2)
    
    # PAUSE 버튼 그리기
    cv2.rectangle(base_canvas, (220, 680), (360, 750), (0, 140, 255), -1)
    cv2.putText(base_canvas, "PAUSE", (245, 720), cv2.FONT_HERSHEY_SIMPLEX, 0.8, (255, 255, 255), 2)
    
    # EXIT 버튼 그리기
    cv2.rectangle(base_canvas, (400, 680), (540, 750), (50, 50, 200), -1)
    cv2.putText(base_canvas, "EXIT", (440, 720), cv2.FONT_HERSHEY_SIMPLEX, 0.8, (255, 255, 255), 2)
    # 이제 이 'base_canvas'는 변하지 않는 UI 도장이 되었습니다. 매 프레임마다 그리지 않고 이 도장을 찍어서 씁니다.
    # -----------------------------------------------------------------------------

    prev_time = 0      # FPS 계산을 위해 이전 프레임의 시간을 저장하는 변수
    last_send_time = 0 # 5초 타이머를 재기 위해 마지막으로 서버에 전송한 시간을 기억하는 변수
    fps = 0            # 현재 측정된 FPS
    combined_view = None # 최종적으로 완성된 카메라 영상을 담을 변수

    # ====== 실시간 반복(루프) 시작 ======
    # 1초에 이 작업을 10~30번씩 반복하여 동영상처럼 보이게 합니다.
    while is_running:
        # 만약 사용자가 강제로 창 우측 상단의 'X' 버튼을 눌러 창을 닫으면 반복을 멈춥니다.
        if cv2.getWindowProperty(window_name, cv2.WND_PROP_VISIBLE) < 1:
            break

        # 일시정지(PAUSED) 상태가 아닐 때만 카메라에서 새 장면을 가져옵니다.
        if not is_paused:
            # 카메라 찰칵! (ret이 False면 카메라 선이 뽑혔거나 에러가 난 것임)
            ret, frame = cap.read()
            if not ret: break

            # [최적화 3 적용] 위에서 만들어둔 '지도(Map)'를 보고 픽셀들만 샥샥 제자리로 이동시킵니다.
            # 복잡한 수학 계산(undistort)을 하는 것보다 컴퓨터 입장에서 압도적으로 빠릅니다.
            if has_calibration:
                frame = cv2.remap(frame, map1, map2, interpolation=cv2.INTER_LINEAR)

            # 영상을 캔버스 윗부분에 넣기 위해 1152x648 크기로 늘립니다(1.2배 확대).
            frame_resized = cv2.resize(frame, (1152, 648))

            # 트랙바에서 소프트웨어 밝기 값을 읽어옵니다. (0~200 값을 가져와서 100을 빼면 -100~100이 됩니다)
            trackbar_pos = cv2.getTrackbarPos('S/W Brightness (-100~100)', window_name)
            actual_brightness = trackbar_pos - 100 

            # 색상 반전 오류를 막기 위해 밝게 할 때(add)와 어둡게 할 때(subtract)를 완벽히 나눕니다.
            if actual_brightness > 0:
                # 영상과 똑같은 크기의 '더하기용 판'을 만들고, 안전한 cv2.add로 더합니다. (255를 넘으면 무조건 255에서 멈춤)
                brightness_array = np.full(frame_resized.shape, actual_brightness, dtype=np.uint8)
                filtered_frame = cv2.add(frame_resized, brightness_array)
            elif actual_brightness < 0:
                # 영상과 똑같은 크기의 '빼기용 판'을 만들고, 안전한 cv2.subtract로 뺍니다. (0보다 낮아지면 0에서 멈춤)
                brightness_array = np.full(frame_resized.shape, abs(actual_brightness), dtype=np.uint8)
                filtered_frame = cv2.subtract(frame_resized, brightness_array)
            else:
                # 트랙바가 정중앙(0)이면 아무것도 하지 않고 원본을 유지합니다.
                filtered_frame = frame_resized.copy()

            # 단축키 'f'를 눌러 필터가 켜진 상태라면, 위에서 만든 CLAHE(그림자 살리기) 필터를 입혀줍니다.
            if use_glare_filter:
                filtered_frame = apply_clahe(filtered_frame)
                
            # 모든 보정이 끝난 깨끗한 장면을 변수에 저장합니다.
            combined_view = filtered_frame.copy()

            # 현재 시간을 측정하여 1초에 장면이 몇 번 도는지(FPS) 계산합니다.
            curr_time = time.time()
            fps = 1 / (curr_time - prev_time) if prev_time > 0 else 0
            
            # [최적화 1 적용] 마지막으로 전송한 지 5초가 지났는지 확인합니다.
            if curr_time - last_send_time >= 5:
                # 기존처럼 함수를 직접 부르면, 서버 응답이 올 때까지 프로그램 전체가 '정지'합니다. (뚝뚝 끊기는 원인)
                # 그래서 'threading(심부름꾼)'을 고용해 데이터 전송만 맡기고, 메인 프로그램은 곧바로 다음 프레임 작업으로 넘어갑니다.
                # daemon=True: 메인 창이 닫히면 심부름을 가던 녀석도 같이 바로 죽도록(종료) 설정합니다.
                threading.Thread(target=send_inspection_result, args=("정상",), daemon=True).start()
                last_send_time = curr_time

            # 현재 시간을 과거 시간으로 기억해둡니다. (다음 프레임 FPS 계산용)
            prev_time = curr_time

        # combined_view에 영상이 잘 들어있다면 이제 화면에 보여줄 준비를 합니다.
        if combined_view is not None:
            # [최적화 2 적용] 매번 버튼 3개와 검은 배경을 그리는 헛수고를 하지 않습니다.
            # 시작할 때 만들어둔 도장(base_canvas)을 찰칵 복사해서 깨끗한 배경판을 바로 가져옵니다.
            canvas = base_canvas.copy()

            # 캔버스의 위쪽(0~648 범위)에 방금 촬영/보정된 영상을 덮어씌웁니다.
            canvas[0:648, 0:1152] = combined_view

            # 4. 버튼은 복사해왔지만 실시간으로 값이 변하는 텍스트(숫자)들은 매번 새로 그려줘야 합니다.
            filter_status = "ON" if use_glare_filter else "OFF"
            
            # 현재 조절 바의 숫자를 읽어와 UI에 글씨로 보여줄 준비를 합니다.
            trackbar_pos = cv2.getTrackbarPos('S/W Brightness (-100~100)', window_name)
            display_brightness = trackbar_pos - 100 
            
            try:
                exposure_pos = cv2.getTrackbarPos('Lens Exposure (-10~0)', window_name)
                display_exposure = exposure_pos - 10
            except:
                display_exposure = "Auto" # 초기화 전 에러 방지용
            
            # 실시간 변동 수치들을 캔버스 우측 하단에 그려줍니다.
            cv2.putText(canvas, f"LIVE | FPS: {int(fps)}", (780, 685), cv2.FONT_HERSHEY_SIMPLEX, 0.65, (0, 255, 0), 2)
            cv2.putText(canvas, f"Brightness: {display_brightness}", (780, 715), cv2.FONT_HERSHEY_SIMPLEX, 0.65, (255, 200, 0), 2)
            cv2.putText(canvas, f"Lens Exposure: {display_exposure}", (780, 745), cv2.FONT_HERSHEY_SIMPLEX, 0.65, (200, 200, 255), 2)
            cv2.putText(canvas, f"Filter(f): {filter_status}", (780, 775), cv2.FONT_HERSHEY_SIMPLEX, 0.65, (0, 255, 255) if use_glare_filter else (150, 150, 150), 2)

            # 일시정지 상태라면, 캔버스 한가운데에 빨간색 경고 문구를 크게 띄웁니다.
            if is_paused:
                cv2.putText(canvas, "--- PAUSED ---", (400, 324), cv2.FONT_HERSHEY_SIMPLEX, 1.5, (0, 0, 255), 4)

            # 최종 완성된 합성 이미지(영상+버튼+텍스트)를 윈도우 창에 띄워서 사용자에게 보여줍니다.
            cv2.imshow(window_name, canvas)

        # 키보드 입력을 감지합니다. 1ms 동안 기다렸다가 무슨 키가 눌렸는지 확인합니다.
        key = cv2.waitKey(1) & 0xFF
        if key == ord('q'):     # 'q' 키를 누르면 while 루프를 빠져나갑니다(종료).
            break
        elif key == ord('f'):   # 'f' 키를 누르면 빛 반사 필터(CLAHE)를 껐다 켤 수 있습니다. (토글)
            use_glare_filter = not use_glare_filter

    # ==== 루프(프로그램) 종료 ====
    print("[SYSTEM] App closed safely.")
    if cap is not None:
        cap.release() # 카메라 장치를 안전하게 풀어줍니다(다른 프로그램이 쓸 수 있도록).
    cv2.destroyAllWindows() # 열려있던 모든 윈도우 창을 닫습니다.

# 파이썬에서 이 파일을 직접 실행했을 때만 main() 함수를 실행하도록 하는 약속된 문법입니다.
if __name__ == "__main__":
    main()
